#include <stdio.h>

int main() {
    int num1, num2, difference;

    // Input two numbers
    printf("Enter first number: ");
    scanf("%d", &num1);

    printf("Enter second number: ");
    scanf("%d", &num2);

    // Perform subtraction
    difference = num1 - num2;

    // Display the result
    printf("The difference of %d and %d is %d\n", num1, num2, difference);

    return 0;
}
